	private Date from;

	/**
	 * Determines if a model is active. A model is active if now is after or
	 * equal to from , and thru is either null, or after now, or equal to now.
	 */
	@Transient

		Date now = new Date();
		boolean afterFrom = from.before(now) || from.equals(now);
